# Verificando se os modelos esto sofrendo de overfitting ou underfitting

# Validao cruzada
scores_dt = cross_val_score(dt, X, y, cv=5, scoring='f1_macro')
scores_rf = cross_val_score(rf, X, y, cv=5, scoring='f1_macro')

print('F1-score mdio rvore de Deciso:', np.mean(scores_dt))
print('F1-score mdio Floresta Aleatria:', np.mean(scores_rf))

# Verificando atravs de grficos (elbow curve) se esto ocorrendo overfitting ou underfitting
inertia = []
for i in range(1, 11):
    kmeans = KMeans(n_clusters=i, random_state=42)
    kmeans.fit(X)
    inertia.append(kmeans.inertia_)
plt.figure(figsize=(10, 6))
plt.plot(range(1, 11), inertia, marker='o')
plt.title('Elbow Curve')
plt.xlabel('Nmero de Clusters')
plt.ylabel('Inertia')
plt.show()
